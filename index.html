async function submitRegistration() { const btn = document.getElementById('submit-btn'); const doctorName = document.getElementById('name-list').value; const card = document.getElementById('main-card'); if (!doctorName) { btn.classList.add('crimson-warning'); btn.innerText = "CHOOSE A NAME FIRST!"; setTimeout(() => { btn.classList.remove('crimson-warning'); btn.innerText = "UPDATE"; }, 2000); return; } card.classList.add('loading-active'); btn.innerText = "UPLOADING..."; try { await fetch(scriptUrl, { method: 'POST', mode: 'no-cors', body: JSON.stringify({ action: "submitDuty", specialty: document.getElementById('spec-select').value, name: doctorName }) }); // 1. Stop loading animation card.classList.remove('loading-active'); // 2. Show Success state btn.classList.add('fetched-active'); btn.innerText = "SUCCESS ✅"; // 3. REVERT BUTTON ONLY (Keep the name and specialty selected) setTimeout(() => { btn.classList.remove('fetched-active'); btn.innerText = "UPDATE"; }, 2000); } catch (e) { card.classList.remove('loading-active'); btn.innerText = "RETRY"; setTimeout(() => { btn.innerText = "UPDATE"; }, 2000); } }
