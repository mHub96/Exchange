async function submitRegistration() { const btn = document.getElementById('submit-btn'); const specSelect = document.getElementById('spec-select'); // Reference to specialty dropdown const doctorName = document.getElementById('name-list').value; const card = document.getElementById('main-card'); if (!doctorName) { btn.classList.add('crimson-warning'); btn.innerText = "CHOOSE A NAME FIRST!"; setTimeout(() => { btn.classList.remove('crimson-warning'); btn.innerText = "UPDATE"; }, 2000); return; } card.classList.add('loading-active'); btn.innerText = "UPLOADING..."; try { await fetch(scriptUrl, { method: 'POST', mode: 'no-cors', body: JSON.stringify({ action: "submitDuty", specialty: specSelect.value, name: doctorName }) }); card.classList.remove('loading-active'); btn.classList.add('fetched-active'); btn.innerText = "SUCCESS ✅"; // RESET SELECTION TO PLACEHOLDER specSelect.value = ""; toggleNameSection(); // This hides the name list and resets its content setTimeout(() => { btn.classList.remove('fetched-active'); btn.innerText = "UPDATE"; }, 2000); } catch (e) { card.classList.remove('loading-active'); btn.innerText = "RETRY"; } }
